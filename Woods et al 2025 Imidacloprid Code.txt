

/*Leaf level data follows. Not all of the arthropods observed on leaves were reported in the manuscript.
Variables are:
Year: 2021, 2022, and 2023
Date: MM/DD/YY format
Block: Replicate block indicted with Roman numerals I to VI for the six replicate blocks
TRT: Treatment, where 1 = nontreated, 2 = imidacloprid applied in May, and 3 = imidacloprid applied in June
Height: Height in the canopy where leaves were collected, either Lower (<2 m) or Upper (>2 m)
Leaf: Subsampled leaves collected from each height x block x trt
TSSM: Motile Tetranychus urticae
TSSM: Tetranychus eggs
PredM: Motile Phytoseiidae mites
PredE: Phytoseiidae eggs
StethL: Larvae of Stethorus spp.
StethE: Eggs of Stethorus spp.
Aphid: Aphids, primarily Phorodon humuli
Anystid: Motile Anystidae mites
Values of arthropods are counts per leaf, with the exception of select dates in 2023 where arthropods were collected with the aid of a mite brushing machine.
On dates when a mite brushing machine was used, a single value is included for Leaf 1, which represents the mean number of arthropods recovered per leaf in the sample for that experimental unit.
*/

data Provado;
input Year	Date $	Block $	TRT	Height $ Leaf	TSSM	TSSME	PredM	PredE	StethL	StethE	Aphid	Anystid;
cards;

;

*Sort data for later manipulation;
proc sort data=Provado;
by Year Height Date;
run;
/* Preliminary visualizations for initial quality assurance check are coded out here;
proc sgpanel data=Provado;
panelby Year Height/columns=6;
vbox TSSM/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Provado;
panelby Year Height/columns=6;
vbox TSSME/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Provado;
panelby Year Height/columns=6;
vbox PredM/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;
*/

*Calculation of means over samples per Year x Height x TRT x Date x Block; 
proc means data=Provado noprint;
class Year Height TRT Date Block;
var TSSM TSSME;
output out=out1 mean(TSSM)=TSSM mean(TSSME)=TSSME std(TSSM)=StDev_Mites std(TSSME)=StDev_Eggs;
run;

/*
Data check on outputted data.
proc print data=out1;
where _type_=31;
run;
*/

*Mixed models for log10 transformed means follows. The best fitting models for each year are given.;
proc glimmix data=out1 plots=residualpanel;
tTSSM=log10(TSSM+1);
where _type_=31 AND Year =2022;
by Year Height;
class TRT Date Block;
model tTSSM=TRT|Date/ddfm=kr dist=normal;
*random Block; *Block effect covariance is 0 for 2022, removed;
random _residual_/sub=TRT*Block type=cs; *Model could not be estimated without CS for residuals;
lsmeans TRT Date TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;


proc glimmix data=out1 plots=residualpanel;
tTSSM=log10(TSSM+1);
where _type_=31 AND Year ne 2022;
by Year Height;
class TRT Date Block;
model tTSSM=TRT|Date/ddfm=kr dist=normal;
*random Block; *Block effect covariance is 0 for 2021 and 2023 at all heights except lower 2023;
random _residual_/sub=TRT*Block type=csh; *AR(1), cs, and csh test--csh has lowest AIC;
lsmeans TRT TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;
/* *Analysis of mite eggs; *Qualitatively similar to TSSM, so omitting from output;
proc glimmix data=out1 plots=residualpanel;
tTSSME=log10(TSSME+1);
where _type_=31;
by Year Height;
class TRT Date Block;
model tTSSME=TRT|Date/ddfm=kr dist=normal;
random Block;
random _residual_/sub=TRT*Block type=csh;
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;
*/

/*Data for canopy shake samples follows;
Year: 2021, 2022, and 2023
Date: MM/DD/YY format
Block: Replicate block indicted with Roman numerals I to VI for the six replicate blocks
TRT: Treatment, where 1 = nontreated, 2 = imidacloprid applied in May, and 3 = imidacloprid applied in June
Shakes: Number of bines from which shake samples were collected.
LBAdult: Aphidphagous ladybeetle adults, all species
Psyllabora: Psyllabora, typically adults
LBL: Aphidphagous ladybeetle larvae, all species
StethA: Adults of Stethorus spp.
StethL: Larvae of Stethorus spp.
MPA: Minute pirate bug adults, Orius spp.
MPN: Minute pirate bug larve, Orius spp.
Hoverfly: Syrphid spp.
LWL: Lacewing larvae
Spiders: All spiders, not identified
Anystid: Motile Anystidae mites
Hymenop: Hymenoptera, not identified
Diabrotica: Diabrotica undecimpunctata adult
DBANymph: Deraeocoris brevis, adult or nymph
Nabid: Nabidae (damsel bugs) adult or nymph
LooperL: Hop looper larvae
Hempitera: Other predatory hemiptera
Striped Thrips: Aeolothrips adults or nymph
Data for each are a count summed over the shake samples collected within a plot 
*/

data Shake;
input Year	Date $	Rep $ TRT	Shakes	LBAdult	Psyllabora	LBL	StethA	StethL	MPA	MPN	HoverFly	LWL	Spiders	Anystid	Hymenop	Diabrotica	DBANymph	Nabid	LooperL	Hemiptera	StripedThrips;
MacroPred=(StethA +	StethL + MPA + MPN	+ DBANymph + Nabid	+ Hemiptera + StripedThrips)/Shakes;
LB=(LBAdult + LBL)/Shakes;
MP=(MPA + MPN)/Shakes;
Steth=(StethA + StethL)/Shakes;
Looper=(LooperL)/Shakes;
cards;


;

proc sort data=Shake;
by Year Date TRT;
run;

proc means data=Shake noprint;
class Year TRT Date Rep;
var Steth MP MacroPred;
output out=out2 mean(Steth)=Steth mean(MP)=MP std(Steth)=StDev_Steth std(MP)=StDev_MP mean(MacroPred)=MacroPred std(MacroPred)=StDev_MacroPred;
run;
proc print data=out2; run;

proc glimmix data=out2 plots=residualpanel;
*tSteth=log10(Steth+1); *not transformed due to reasonable residual diagnostics;
where _type_=15 AND Year ne 2023;
by Year;
class TRT Date Rep;
model Steth=TRT|Date/ddfm=kr dist=normal;
*random Rep; *Replicate block removed due to 0 covariance parameter estimate in 2021 and 2022;
random _residual_/sub=TRT*Rep type=cs; *AIC is lower with CS vs CSH;
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;
proc glimmix data=out2 plots=residualpanel;
*tSteth=log10(Steth+1); *not transformed due to reasonable residual diagnostics;
where _type_=15 AND Year = 2023;
by Year;
class TRT Date Rep;
model Steth=TRT|Date/ddfm=kr dist=normal;
random Rep;
random _residual_/sub=TRT*Rep type=csh; *csh is superior to CS for 2023;
lsmeans TRT*Date/lines  slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;
proc glimmix data=out2 plots=residualpanel;
*tMP=log10(MP+1); *not transformed due to reasonable residual diagnostics;
where _type_=15;
by Year;
class TRT Date Rep;
model MP=TRT|Date/ddfm=kr dist=normal;
*random Rep; *0 covariance parameter estimates for replicate blocl;
random _residual_/sub=TRT*Rep type=csh; *csh has lowest AIC vs. cs and ar(1);
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;

proc glimmix data=out2 plots=residualpanel;
tMacroPred=log10(MacroPred+1); *not transformed due to reasonable residual diagnostics;
where _type_=15 AND Year = 2021;
by Year;
class TRT Date Rep;
model tMacroPred=TRT|Date/ddfm=kr dist=normal;
*random Rep; *0 covariance parameter estimates for replicate block;
random _residual_/sub=TRT*Rep type=cs; *csh has lowest AIC vs. cs and ar(1);
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;

proc glimmix data=out2 plots=residualpanel;
tMacroPred=log10(MacroPred+1); *not transformed due to reasonable residual diagnostics;
where _type_=15 AND Year = 2022;
by Year;
class TRT Date Rep;
model tMacroPred=TRT|Date/ddfm=kr dist=normal;
random Rep; *0 covariance parameter estimates for replicate block;
random _residual_/sub=TRT*Rep type=cs; *csh has lowest AIC vs. cs and ar(1);
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;

proc glimmix data=out2 plots=residualpanel;
tMacroPred=log10(MacroPred+1); *not transformed due to reasonable residual diagnostics;
where _type_=15 AND Year = 2023;
by Year;
class TRT Date Rep;
model tMacroPred=TRT|Date/ddfm=kr dist=normal;
*random Rep; *0 covariance parameter estimates for replicate block;
random _residual_/sub=TRT*Rep type=ar(1); *csh has lowest AIC vs. cs and ar(1);
lsmeans TRT*Date/lines slicediff=Date plots=meanplot(sliceby=trt plotby=date cl);
run;
/* *Data visualizations for preliminary quality assurance check;

proc sgpanel data=Shake;
panelby Year/columns=3;
vbox LB/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Shake;
panelby Year/columns=3;
vbox Steth/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Shake;
panelby Year/columns=3;
vbox MP/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Shake;
panelby Year/columns=3;
vbox Looper/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;

proc sgpanel data=Shake;
panelby Year/columns=3;
vbox MacroPred/category=TRT group=date;
*scatter x=date y=TSSM/group=TRT;
run;
*/	

/*Yield data follows:
Year: 2021, 2022, and 2023
Date: Specific date of harvest in MM/DD/YY format
Block: Replicate block indicted with Roman numerals I to VI for the six replicate blocks
Treatment: Imidacloprid treatment, where 1 = nontreated, 2 = imidacloprid applied in May, and 3 = imidacloprid applied in June
Strings: Number of strings harvested per plot
DM: Dry matter, expressed as a proportion
_Alpha: Percent alpha-acids, standardized to 8% moisture
_Beta: Percent beta-aicds, standardized to 8% moisture
HSI: Hop Storage Index
Dry_Yield: Yield in kg per string
Alpha_Yield: Yield of alpha-acids in kg per string
Color: 1 to 10 ordinal value for cone color
*/



data Yield;
input Year	Date $	Block $	Treatment	Strings	DM	_Alpha	_Beta	HSI	Dry_Yield	Alpha_Yield	Color;
cards;


;
/* *Data visualization for preliminary check;
proc sgpanel data=Yield;
panelby Year/columns=3;
vbox DM/category=Treatment;
run;

proc sgpanel data=Yield;
panelby Year/columns=3;
vbox _Alpha/category=Treatment;
run;

proc sgpanel data=Yield;
panelby Year/columns=3;
vbox _Beta/category=Treatment;
run;

proc sgpanel data=Yield;
panelby Year/columns=3;
vbox HSI/category=Treatment;
run;

proc sgpanel data=Yield;
panelby Year/columns=3;
vbox Dry_Yield/category=Treatment;
run;

proc sgpanel data=Yield;
panelby Year/columns=3;
vbox Color/category=Treatment;
run;
*/

proc glimmix data=Yield plots=residualpanel nobound; 
by Year;
Class Treatment Block;
model DM=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;

proc glimmix data=Yield plots=residualpanel nobound;
by Year;
Class Treatment Block;
model _Alpha=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;

proc glimmix data=Yield plots=residualpanel nobound;
by Year;
Class Treatment Block;
model _Beta=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;

proc glimmix data=Yield plots=residualpanel nobound;
by Year;
Class Treatment Block;
model HSI=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;

proc glimmix data=Yield plots=residualpanel nobound;
by Year;
Class Treatment Block;
model Dry_Yield=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;

by Year;
Class Treatment Block;
model Color=Treatment;
random Block;
lsmeans Treatment/lines cl;
run;
